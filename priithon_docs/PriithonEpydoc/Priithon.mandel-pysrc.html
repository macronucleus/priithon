 13 -def mandel(ny=512, nx=512, itermax=30, ymin=-1.25, ymax=1.25, xmin=-2, xmax=.6): 

 14      """ 
 15      Fast mandelbrot computation using numpy. 
 16   
 17      (nx, ny) are the output image dimensions 
 18      x: real axis 
 19      y: imaginary axis 
 20   
 21      itermax is the maximum number of iterations to do 
 22      ymin, ymax, xmin, xmax specify the region of the 
 23      set to compute. 
 24      """ 
 25      # The point of iy and ix is that they are 2D arrays 
 26      # giving the y-coord and x-coord at each point in 
 27      # the array. The reason for doing this will become 
 28      # clear below... 
 29      iy, ix = mgrid[0:ny, 0:nx] 
 30   
 31      # Now y and x are the y-values and x-values at each 
 32      # point in the array, linspace(start, end, n) 
 33      # is an array of n linearly spaced points between 
 34      # start and end, and we then index this array using 
 35      # numpy fancy indexing. If A is an array and I is 
 36      # an array of indices, then A[I] has the same shape 
 37      # as I and at each place i in I has the value A[i]. 
 38      y = linspace(ymin, ymax, ny)[iy] 
 39      x = linspace(xmin, xmax, nx)[ix] 
 40   
 41      # c is the complex number with the given y, x coords 
 42      c = x+complex(0,1)*y 
 43   
 44      del y, x # save a bit of memory, we only need z 
 45   
 46      # the output image coloured according to the number 
 47      # of iterations it takes to get to the boundary 
 48      # abs(z)>2 
 49      img = zeros(c.shape, dtype=int) 
 50      # Here is where the improvement over the standard 
 51      # algorithm for drawing fractals in numpy comes in. 
 52      # We flatten all the arrays iy, ix and c. This 
 53      # flattening doesn't use any more memory because 
 54      # we are just changing the shape of the array, the 
 55      # data in memory stays the same. It also affects 
 56      # each array in the same way, so that index i in 
 57      # array c has y, x coords iy[i], ix[i]. The way the 
 58      # algorithm works is that whenever abs(z)>2 we 
 59      # remove the corresponding index from each of the 
 60      # arrays iy, ix and c. Since we do the same thing 
 61      # to each array, the correspondence between c and 
 62      # the y, x coords stored in iy and ix is kept. 
 63      iy.shape = \ 
 64          ix.shape = \ 
 65          c.shape = ny*nx 
 66   
 67      # we iterate z->z^2+c with z starting at 0, but the 
 68      # first iteration makes z=c so we just start there. 
 69      # We need to copy c because otherwise the operation 
 70      # z->z^2 will send c->c^2. 
 71      z = copy(c) 
 72   
 73      for i in xrange(itermax): 
 74          if not len(z): break # all points have escaped 
 75   
 76          # equivalent to z = z*z+c but quicker and uses 
 77          # less memory 
 78          multiply(z, z, z) 
 79          add(z, c, z) 
 80   
 81          # these are the points that have escaped 
 82          rem = abs(z)>2.0 
 83   
 84          # colour them with the iteration number, we 
 85          # add one so that points which haven't 
 86          # escaped have 0 as their iteration number, 
 87          # this is why we keep the arrays iy and ix 
 88          # because we need to know which point in img 
 89          # to colour 
 90          img[iy[rem], ix[rem]] = i+1 
 91   
 92          # -rem is the array of points which haven't 
 93          # escaped, in numpy -A for a boolean array A 
 94          # is the NOT operation. 
 95          rem = -rem 
 96   
 97          # So we select out the points in 
 98          # z, iy, ix and c which are still to be 
 99          # iterated on in the next step 
100          z = z[rem] 
101          iy, ix = iy[rem], ix[rem] 
102          c = c[rem] 
103   
104      return img
